    \item Female types $\omega$

    \item Masses $N_m, N_w$
    \item Entry Flows $\lambda_m, \lambda_w$
    \item Tightness $\tau = \min\{\frac{N_w}{N_m}, 1\}$
    \item Effective discount $\alpha$ 
\end{itemize}
